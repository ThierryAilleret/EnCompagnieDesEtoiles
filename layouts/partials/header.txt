<script>
		/* Pour le panier */
		function ouvrirOuFermerPanier() {
			const volet = document.getElementById('mini-panier');
			const indicateur = document.getElementById('panier-toggle');

			if (volet.classList.contains('actif')) {
				// üîí Volet d√©j√† ouvert ‚Üí on le ferme
				volet.classList.remove('actif');
				if (indicateur) indicateur.classList.remove('panier-decale');
			} else {
				// üîì Volet ferm√© ‚Üí on l‚Äôouvre
				volet.classList.add('actif', 'slide-in');
				if (indicateur) indicateur.classList.add('panier-decale');
				afficherMiniPanier();
				setTimeout(() => volet.classList.remove('slide-in'), 300);
			}
		}



		function fermerPanier() {
			const volet = document.getElementById('mini-panier');
			const indicateur = document.getElementById('panier-toggle');

			volet.classList.remove('actif');
			if (indicateur) indicateur.classList.remove('panier-decale');
		}

		function afficherMiniPanier() {
			const panier = JSON.parse(localStorage.getItem("panier")) || [];
			const conteneur = document.getElementById("contenu-panier");
			conteneur.innerHTML = "";

			let total = 0;

			panier.forEach(item => {
				total += parseFloat(item.prix) * item.quantite;
				const bloc = document.createElement("div");
				bloc.className = "item-panier";
				bloc.innerHTML = `
					<img src="${item.image}" alt="${item.titre}">
					<div class="info-item">
						<strong>${item.titre}</strong><br>
						${item.quantite} √ó ${item.prix} ‚Ç¨
						<div class="quantite-controls">
							<button onclick="modifierQuantite('${item.titre}', 1)">+</button>
							<button onclick="modifierQuantite('${item.titre}', -1)">‚àí</button>
						</div>
						<button class="btn-supprimer" onclick="supprimerDuPanier('${item.titre}')" aria-label="Retirer du panier">
							üóëÔ∏è
						</button>
					</div>
				`;
				conteneur.appendChild(bloc);
			});

			document.getElementById("total-panier").textContent = `Total : ${total.toFixed(2)} ‚Ç¨ hors frais de port`;

			const btnCommander = document.querySelector(".btn-commander");
			if (btnCommander) {
				if (panier.length === 0) {
					btnCommander.disabled = true;
					btnCommander.classList.add("desactive");
				} else {
					btnCommander.disabled = false;
					btnCommander.classList.remove("desactive");
				}
			}

		}

		function modifierQuantite(titre, delta) {
			const panier = JSON.parse(localStorage.getItem("panier")) || [];
			const index = panier.findIndex(p => p.titre === titre);
			if (index >= 0) {
				panier[index].quantite += delta;
				if (panier[index].quantite <= 0) {
					panier.splice(index, 1);
				}
				localStorage.setItem("panier", JSON.stringify(panier));
				window.dispatchEvent(new Event("panierMisAJour"));
				afficherMiniPanier();
				mettreAJourIndicateur(); // üëà mise √† jour du chiffre
			}
			const indicateurEl = document.getElementById("panier-toggle");
			if (indicateurEl) {
				indicateurEl.classList.add("pulse");
				setTimeout(() => indicateurEl.classList.remove("pulse"), 300);
			}
		}


		function supprimerDuPanier(titre) {
			const panier = JSON.parse(localStorage.getItem("panier")) || [];
			const nouveau = panier.filter(p => p.titre !== titre);
			localStorage.setItem("panier", JSON.stringify(nouveau));
			window.dispatchEvent(new Event("panierMisAJour"));
			afficherMiniPanier();
			mettreAJourIndicateur(); // üëà mise √† jour du chiffre
			
			const indicateurEl = document.getElementById("panier-toggle");
			if (indicateurEl) {
				indicateurEl.classList.add("pulse");
				setTimeout(() => indicateurEl.classList.remove("pulse"), 300);
			}

		}

		document.addEventListener("click", function (event) {
			const volet = document.getElementById("mini-panier");
			const toggle = document.getElementById("panier-toggle");

			// Bouton "Ajouter au panier"
			const ajouterBtn = event.target.closest(".btn-add-to-cart");

			// Boutons quantit√© et corbeille
			const quantiteBtn = event.target.closest(".quantite-controls");
			const supprimerBtn = event.target.closest(".btn-supprimer");

			if (!volet || !volet.classList.contains("actif")) return;

			if (
				volet.contains(event.target) ||
				toggle.contains(event.target) ||
				ajouterBtn ||
				quantiteBtn ||
				supprimerBtn
			) {
				return; // üëâ ne rien faire si l‚Äô√©l√©ment est dans la liste
			}

			// Sinon ‚Üí fermeture du volet
			fermerPanier();
		});


		function mettreAJourIndicateur() {
			const panier = JSON.parse(localStorage.getItem("panier")) || [];
			const total = panier.reduce((sum, item) => sum + item.quantite, 0);

			const compteEl = document.getElementById("panier-compte");
			const indicateurEl = document.getElementById("panier-toggle");

			if (compteEl) compteEl.textContent = total;

			if (indicateurEl) {
				if (total === 0) {
					indicateurEl.style.background = "white";
					indicateurEl.style.color = "black";
				} else {
					indicateurEl.style.background = "black";
					indicateurEl.style.color = "white";
				}
			}
		}

		window.addEventListener("pageshow", function () {
			mettreAJourIndicateur();
		});

</script>
